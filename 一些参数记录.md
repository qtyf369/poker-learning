pygame.draw.rect(
    surface,    # 绘制目标（比如窗口 screen）
    color,      # 颜色（RGB元组，比如 (200,200,200) 灰色）
    rect,       # 矩形信息（可以是 (x,y,宽,高) 元组，或 pygame.Rect 对象）
    width=0,    # 可选：线宽（0=填充矩形，>0=只画边框）
    border_radius=0,  # 可选：边角圆角半径（0=直角，>0=圆角）
    border_top_left_radius=-1,  # 可选：单独设置左上角圆角（-1=用 border_radius）
    border_top_right_radius=-1, # 可选：单独设置右上角圆角
    border_bottom_left_radius=-1, # 可选：单独设置左下角圆角
    border_bottom_right_radius=-1 # 可选：单独设置右下角圆角
)
